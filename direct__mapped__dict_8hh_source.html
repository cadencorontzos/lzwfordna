32// Direct Mapped Dictionary:
33//
34// We will have dictionaries for every length 1 to MAX_STRING_LENGTH, for
35// strings of that length
36//
37// We will map the 4 neucleotides into two bits, so any string of length n can
38// be represented by 2n bits That 2n bit encoding is an index into the nth
39// dictionary
40//
41// in the nth dictionary, there is space for every string of length n
42//
43// upon lookup, we look at the space in the nth dictionary corresponding to the
44// 2n bit index given by our lookup string if there is a codeword there, we
45// return it. If the codeword is 0, we know we haven't seen that string before
46//
47// for strings longer than length MAX_STRING_LENGTH, we have a unordered_map on
48// top to take the rest
49//
50// when we run out of codewords, we simply don't allow any more strings to be
51// added
52//
53class LZW_Encode_Dictionary
54 : private LZWDictionary<codeword_type, Next_Longest_Run> {
55private:

59 // 00000110 00000110 00000110 00000110 00000110 00000110 00000110 00000110

62 index_type map_str(const char *input, unsigned len) const {
63 index_type result = 0;
64 for (unsigned i = 0; i < len; i++) {
65 result = (result << 2) + values[input[i]];

67 return result;

175 // find longest using binary search.
176 // assumes that the longest run < max
177 // assumes that all characters are in starting dictionary
178 Next_Longest_Run find_longest_binary_search(const char *input, int left,
179 int right,
180 index_type full_index) {
181 int middle = left + ((right - left) / 2);
182 codeword_type entry;
183 index_type index;
184 index_type next_index;
185 while (left <= right) {
186 next_index = full_index >> (INDEX_BITS - (middle + 1) * 2);
187 index = next_index >> 2;
188 entry = code_of_manual(middle, index);
189 if (entry == 0) {
190 right = middle - 1;
191 middle = left + ((right - left) / 2);
192 } else if (middle == MAX_STRING_LENGTH ||
193 code_of_manual(middle + 1, next_index) == 0) {
194 return Next_Longest_Run(middle, entry, index);
195 } else {
196 left = middle + 1;
197 middle = left + ((right - left) / 2);
198 }
199 }
200
201 index = values[input[0]];
202 entry = code_of_manual(1, index);
203 return Next_Longest_Run(1, entry, index);
204 }

209 // start at start and loop up or down
210 if (input + 16 > end_of_input) {

214 const uint64_t *input_ints = reinterpret_cast<const uint64_t *>(input);
215 uint64_t swapped = _bswap64(input_ints[0]);
216 uint64_t swapped2 = _bswap64(input_ints[1]);
217 uint64_t first_half = _pext_u64(swapped, MASK);
218 uint64_t second_half = _pext_u64(swapped2, MASK);
219 index_type next_long_index =
220 (first_half << NEXT_LONGEST_INDEX_SIZE) + second_half;
221
222
223 if(!BINARY_SEARCH){
224 // check the starting length
225 index_type index_for_fl_start = next_long_index >>
226 (INDEX_BITS-(FIND_LONGEST_START*2));
227 int entry = code_of_manual(FIND_LONGEST_START, index_for_fl_start);
228 if(entry == 0){

230 index_for_fl_start);
231 }
232 return find_longest_looping_up(input, end_of_input, FIND_LONGEST_START-1,
233 next_long_index);
234
235 }else{
236
237 // start at start and binary search
238 // check the starting string
239 int index = next_long_index >> (INDEX_BITS - FIND_LONGEST_START * 2);
240 int entry = code_of_manual(FIND_LONGEST_START, index);
241 if (entry == 0) {
