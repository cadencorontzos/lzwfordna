225 Next_Longest_Run find_longest_binary_search(const char *input, int left,
226 int right,
227 index_type full_index) {
228 int middle = left + ((right - left) / 2);
229 codeword_type entry;
230 index_type index;
231 index_type next_index;
232 while (left <= right) {
233 next_index = full_index >> (INDEX_BITS - (middle + 1) * 2);
234 index = next_index >> 2;
235 entry = code_of_manual(middle, index);
236 if (entry == 0) {
237 right = middle - 1;
238 middle = left + ((right - left) / 2);
239 } else if (middle == MAX_STRING_LENGTH ||
240 code_of_manual(middle + 1, next_index) == 0) {
241 return Next_Longest_Run(middle, entry, index);
242 } else {
243 left = middle + 1;
244 middle = left + ((right - left) / 2);
245 }
246 }
247
248 index = values[input[0]];
249 entry = code_of_manual(1, index);
250 return Next_Longest_Run(1, entry, index);

253 Next_Longest_Run find_longest_in_dict(const char *input,
254 const char *end_of_input) override {
255
256 // start at start and loop up or down
257 if (input + 16 > end_of_input) {
258 return find_longest_looping_up_on_fly(input, end_of_input, 0, 0);
259 }
260
261 const uint64_t *input_ints = reinterpret_cast<const uint64_t *>(input);
262 uint64_t swapped = _bswap64(input_ints[0]);
263 uint64_t swapped2 = _bswap64(input_ints[1]);
264 uint64_t first_half = _pext_u64(swapped, MASK);
265 uint64_t second_half = _pext_u64(swapped2, MASK);
266 index_type next_long_index =
267 (first_half << NEXT_LONGEST_INDEX_SIZE) + second_half;
268
269 if (!BINARY_SEARCH) {
270 // check the starting length
271 index_type index_for_fl_start =
272 next_long_index >> (INDEX_BITS - (FIND_LONGEST_START * 2));
273 int entry = code_of_manual(FIND_LONGEST_START, index_for_fl_start);
274 if (entry == 0) {
275 return find_longest_looping_down(FIND_LONGEST_START,
276 index_for_fl_start);
277 }
278 return find_longest_looping_up(input, end_of_input,
279 FIND_LONGEST_START - 1, next_long_index);
280
281 } else {
282
283 // start at start and binary search
284 // check the starting string
285 int index = next_long_index >> (INDEX_BITS - FIND_LONGEST_START * 2);
286 int entry = code_of_manual(FIND_LONGEST_START, index);
287 if (entry == 0) {
288 return find_longest_binary_search(input, 1, FIND_LONGEST_START - 1,
289 next_long_index);
290 }
291 return find_longest_binary_search(input, FIND_LONGEST_START,
292 MAX_STRING_LENGTH, next_long_index);
293 }
